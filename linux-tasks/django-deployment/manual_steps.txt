# Prerequisites
- Python 3.7.2+ and python-pip
- MySQL 8.0.15+

# Steps:
1. python3 -m venv venv
2. source venv/bin/activate
3. pip install -r requirements.txt
4. git clone "https://github.com/Manisha-Bayya/simple-django-project.git"
5. cd simple-django-project/
   pip install -r requirements.txt
6. # open mysql bash
   mysql -u <mysql-user> -p

   # Give the absolute path of the file
   mysql> source ~/simple-django-project/world.sql
   mysql> exit;
7. # open settings file
    vim panorbit/settings.py
 
    # Edit Database configurations with your MySQL configurations.
    # Search for DATABASES section.
    DATABASES = {
        'default': {
            'ENGINE': 'django.db.backends.mysql',
            'NAME': 'world',
            'USER': '<mysql-user>',
            'PASSWORD': '<mysql-password>',
            'HOST': '<mysql-host>',
            'PORT': '<mysql-port>',
        }
    }
    
    # Edit email configurations.
    # Search for email configurations
    EMAIL_USE_TLS = True
    EMAIL_HOST = 'smtp.gmail.com'
    EMAIL_HOST_USER = '<your-email>'
    EMAIL_HOST_PASSWORD = '<your-email-password>'
    EMAIL_PORT = 587
    
    # save the file
8. # Make migrations
   python manage.py makemigrations
   python manage.py migrate
   
   # For search feature we need to index certain tables to the haystack. For that run below command.
   python manage.py rebuild_index
   
   # Run the server
   python manage.py runserver 0:8001
   
   # your server is up on port 8001
   
   Try opening http://localhost:8001 in the browser. Now you are good to go.
